{{- if eq .Values.mode "aio" }}
Observer has been deployed in All-in-One (AIO) mode.

1. Access the Web UI:

{{- if contains "NodePort" .Values.aio.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "observer.fullname" . }}-aio)
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo "Web UI:        http://$NODE_IP:$NODE_PORT"
  echo "gRPC Endpoint: $NODE_IP:$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[1].nodePort}" services {{ include "observer.fullname" . }}-aio)"

{{- else if contains "LoadBalancer" .Values.aio.service.type }}
  NOTE: It may take a few minutes for the LoadBalancer IP to be provisioned.
  Watch: kubectl get --namespace {{ .Release.Namespace }} svc -w {{ include "observer.fullname" . }}-aio

  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "observer.fullname" . }}-aio --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo "Web UI:        http://$SERVICE_IP:{{ .Values.aio.service.webPort }}"
  echo "gRPC Endpoint: $SERVICE_IP:{{ .Values.aio.service.grpcPort }}"

{{- else }}
  # Port-forward the Web UI
  kubectl --namespace {{ .Release.Namespace }} port-forward svc/{{ include "observer.fullname" . }}-aio {{ .Values.aio.service.webPort }}:{{ .Values.aio.service.webPort }}
  # Then visit: http://127.0.0.1:{{ .Values.aio.service.webPort }}

  # Port-forward the gRPC endpoint
  kubectl --namespace {{ .Release.Namespace }} port-forward svc/{{ include "observer.fullname" . }}-aio {{ .Values.aio.service.grpcPort }}:{{ .Values.aio.service.grpcPort }}
  # Then connect to: localhost:{{ .Values.aio.service.grpcPort }}
{{- end }}

  NOTE: Exposing services externally is the operator's responsibility.
  Configure a LoadBalancer, NodePort, or external ingress controller
  appropriate for your cluster environment.

1. Access the Web UI:

{{- if contains "NodePort" .Values.distributed.web.service.type }}

  Watch: kubectl get --namespace {{ .Release.Namespace }} svc -w {{ include "observer.fullname" . }}-web
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "observer.fullname" . }}-web --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")

{{- else }}
  kubectl --namespace {{ .Release.Namespace }} port-forward svc/{{ include "observer.fullname" . }}-web 8080:{{ .Values.distributed.web.service.port }}

  kubectl --namespace {{ .Release.Namespace }} port-forward svc/{{ include "observer.fullname" . }}-api 8080:{{ .Values.distributed.api.service.port }}
  # Then: http://127.0.0.1:8080
